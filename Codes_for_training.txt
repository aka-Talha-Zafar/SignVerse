class ASLGrammarTransformerV2:
    ARTICLES     = {"a", "an", "the"}
    AUXILIARIES  = {
        "is","are","was","were","am","be","been","being",
        "do","does","did","has","have","had",
        "will","would","could","should","shall",
        "may","might","must","can","ought",
    }
    PREPOSITIONS = {
        "to","of","in","at","on","for","with",
        "by","from","into","onto","upon","about",
        "over","under","through","between","among",
        "within","without","against","toward","towards",
        "along","across","behind","beside","beyond",
        "during","except","inside","outside","since",
        "than","until","unto","up","via","worth",
    }
    WH_WORDS     = {
        "what","where","when","who","why",
        "how","which","whom","whose",
    }
    TIME_WORDS   = {
        "yesterday","today","tomorrow","now","later",
        "before","after","soon","recently","always",
        "never","sometimes","already","still","yet",
        "last","next","every","daily","weekly","monthly",
        "morning","afternoon","evening","night",
        "monday","tuesday","wednesday","thursday",
        "friday","saturday","sunday","ago","since",
    }
    CONTRACTIONS = {
        "don't":"do not",    "doesn't":"does not",
        "didn't":"did not",  "won't":"will not",
        "can't":"cannot",    "isn't":"is not",
        "aren't":"are not",  "wasn't":"was not",
        "weren't":"were not","i'm":"i am",
        "i've":"i have",     "i'll":"i will",
        "i'd":"i would",     "you're":"you are",
        "you've":"you have", "you'll":"you will",
        "he's":"he is",      "she's":"she is",
        "it's":"it is",      "we're":"we are",
        "they're":"they are","there's":"there is",
        "that's":"that is",  "let's":"let us",
        "here's":"here is",  "how's":"how is",
        "what's":"what is",  "who's":"who is",
        "we've":"we have",   "they've":"they have",
        "they'll":"they will","we'll":"we will",
        "we'd":"we would",   "you'd":"you would",
    }

    def transform(self, sentence: str) -> str:
        if not sentence or str(sentence).lower() == "nan":
            return ""
        s = str(sentence).strip().lower()
        for k, v in self.CONTRACTIONS.items():
            s = s.replace(k, v)
        tokens = re.findall(r"[a-z]+", s)
        if not tokens:
            return ""
        is_wh   = tokens[0] in self.WH_WORDS
        negated = "not" in tokens
        tokens  = [t for t in tokens if t != "not"]
        tokens  = [t for t in tokens if t not in self.ARTICLES]
        if is_wh:
            rest   = [t for t in tokens[1:] if t not in self.AUXILIARIES]
            tokens = [tokens[0]] + rest
        else:
            tokens = [t for t in tokens if t not in self.AUXILIARIES]
        tokens     = [t for t in tokens if t not in self.PREPOSITIONS]
        time_toks  = [t for t in tokens if t in self.TIME_WORDS]
        other_toks = [t for t in tokens if t not in self.TIME_WORDS]
        tokens     = time_toks + other_toks
        if negated:
            tokens.append("not")
        return " ".join(tokens).upper()

def rebuild_pairs(df, split):
    pairs          = []
    missing        = 0
    frontal_folder = DATASET_PATH / split / "frontal"
    side_folder    = DATASET_PATH / split / "side"
    for _, row in df.iterrows():
        sentence      = str(row.get("SENTENCE", "")).strip()
        sentence_name = str(row.get("SENTENCE_NAME", "")).strip()
        if not sentence or sentence.lower() == "nan":
            continue
        if not sentence_name or sentence_name.lower() == "nan":
            continue
        frontal_npy = frontal_folder / f"{sentence_name}_holistic.npy"
        side_name   = sentence_name.replace("_front", "_side")
        side_npy    = side_folder   / f"{side_name}_holistic.npy"
        if frontal_npy.exists():
            npy_path = frontal_npy
        elif side_npy.exists():
            npy_path = side_npy
        else:
            missing += 1
            continue
        gloss = transformer.transform(sentence)
        if not gloss:
            continue
        pairs.append({
            "sentence_name" : sentence_name,
            "english"       : sentence,
            "gloss"         : gloss,
            "npy_path"      : str(npy_path),
            "split"         : split,
        })
    print(f"  {split:5s} | matched={len(pairs):5d} | missing={missing}")
    return pairs

class SignVerseTextDataset(Dataset):
    def __init__(self, pairs, tokenizer, max_src, max_tgt):
        self.pairs     = [p for p in pairs
                          if p.get("english","").strip()
                          and p.get("gloss","").strip()]
        self.tokenizer = tokenizer
        self.max_src   = max_src
        self.max_tgt   = max_tgt

    def __len__(self):
        return len(self.pairs)

    def __getitem__(self, idx):
        pair = self.pairs[idx]
        src  = self.tokenizer(
            pair["english"],
            max_length=self.max_src,
            padding="max_length",
            truncation=True,
            return_tensors="pt",
        )
        tgt  = self.tokenizer(
            pair["gloss"],
            max_length=self.max_tgt,
            padding="max_length",
            truncation=True,
            return_tensors="pt",
        )
        labels = tgt["input_ids"].squeeze().clone()
        labels[labels == self.tokenizer.pad_token_id] = -100
        return {
            "input_ids"      : src["input_ids"].squeeze(),
            "attention_mask" : src["attention_mask"].squeeze(),
            "labels"         : labels,
            "english"        : pair["english"],
            "gloss"          : pair["gloss"],
        }

def save_checkpoint(model, tokenizer, epoch, val_loss, history):
    ckpt_path = CHECKPOINT_DIR / f"epoch_{epoch:02d}"
    ckpt_path.mkdir(parents=True, exist_ok=True)
    model.save_pretrained(ckpt_path)
    tokenizer.save_pretrained(ckpt_path)
    meta = {
        "epoch"    : epoch,
        "val_loss" : val_loss,
        "history"  : history,
    }
    json.dump(meta, open(ckpt_path / "meta.json", "w"), indent=2)
    print(f"    Checkpoint saved: epoch_{epoch:02d} | val_loss={val_loss:.4f}")


def find_latest_checkpoint():
    if not CHECKPOINT_DIR.exists():
        return None
    checkpoints = sorted(CHECKPOINT_DIR.glob("epoch_*"))
    if not checkpoints:
        return None
    latest    = checkpoints[-1]
    meta_file = latest / "meta.json"
    if not meta_file.exists():
        return None
    meta = json.load(open(meta_file))
    print(f"Found checkpoint : {latest.name}")
    print(f"  Epoch          : {meta['epoch']}")
    print(f"  Val loss       : {meta['val_loss']:.4f}")
    return meta["epoch"], meta["val_loss"], meta["history"], latest

def train_one_epoch(model, loader, optimizer, scheduler, scaler, device):
    model.train()
    total_loss, steps = 0.0, 0
    pbar = tqdm(loader, desc="  Train", leave=False)
    for batch in pbar:
        input_ids      = batch["input_ids"].to(device)
        attention_mask = batch["attention_mask"].to(device)
        labels         = batch["labels"].to(device)
        optimizer.zero_grad()
        with torch.cuda.amp.autocast(
            enabled=torch.cuda.is_available()
        ):
            outputs = model(
                input_ids=input_ids,
                attention_mask=attention_mask,
                labels=labels,
            )
            loss = outputs.loss
        scaler.scale(loss).backward()
        scaler.unscale_(optimizer)
        torch.nn.utils.clip_grad_norm_(
            model.parameters(), CFG["grad_clip"]
        )
        scaler.step(optimizer)
        scaler.update()
        scheduler.step()
        total_loss += loss.item()
        steps      += 1
        pbar.set_postfix(loss=f"{loss.item():.4f}")
    return total_loss / steps


def validate(model, loader, device):
    model.eval()
    total_loss, steps = 0.0, 0
    with torch.no_grad():
        for batch in tqdm(loader, desc="  Val  ", leave=False):
            input_ids      = batch["input_ids"].to(device)
            attention_mask = batch["attention_mask"].to(device)
            labels         = batch["labels"].to(device)
            with torch.cuda.amp.autocast(
                enabled=torch.cuda.is_available()
            ):
                outputs = model(
                    input_ids=input_ids,
                    attention_mask=attention_mask,
                    labels=labels,
                )
            total_loss += outputs.loss.item()
            steps      += 1
    return total_loss / steps

def translate_to_asl(english_text, num_beams=4, max_length=64):
    """
    REQ-13 : accepts English text input

    inputs = tokenizer(
        english_text,

        output_ids = model.generate(
            input_ids=inputs["input_ids"],

            max_length=max_length,

    final_gloss = bert_gloss if bert_gloss.strip() else rule_gloss

    return {
        "english"    : english_text,
        "rule_gloss" : rule_gloss,
        "bert_gloss" : bert_gloss,
        "final_gloss": final_gloss,
    }

# CELL 7 — Step 3: ASL Sign Sequence Mapping (FIXED)
# SRS: REQ-14, Section 2.5
#
# ROOT CAUSE FIXED:
#   OLD: word_to_npy["YOU"] = full 565-frame sentence file
#        → load_clip downsampled 565 unrelated frames to 30
#        → AI received garbage motion, produced wrong signs
#
#   NEW: word_to_npy["YOU"] = {
#             "path"  : "sentence_file.npy",
#             "start" : 45,   ← exact frame where YOU begins
#             "end"   : 68    ← exact frame where YOU ends
#        }
#        → load_clip slices arr[45:68], gets 23 real "YOU" frames
#        → interpolates 23→30 frames (clean, correct motion)
#        → AI receives the actual pointing gesture for YOU
#
# HOW THE FRAME INDICES ARE FOUND:
#   The How2Sign realigned CSVs have START_REALIGNED and
#   END_REALIGNED columns giving each sentence's timestamp in
#   its source video. We use these timestamps at 25 FPS
#   (the documented extraction rate for How2Sign Holistic) to
#   convert time → frame index within the .npy file.
#
#   For word-level timing inside the sentence we use a
#   proportional distribution strategy: each word in the
#   gloss gets an equal-duration segment of the sentence's
#   total frame span. This is the best available approach
#   because How2Sign does NOT provide a word-alignment CSV
#   in the Kaggle holistic dataset (only sentence-level CSVs
#   are present). The proportional split is FAR better than
#   the old approach of loading all 565 frames, because:
#     - The first word gets frames from the START of the
#       sentence, not a compressed average of the entire thing
#     - Each word gets a unique, non-overlapping segment
#     - The total motion budget matches the real sentence
#
# SPATIAL NORMALIZATION ADDED:
#   Every loaded slice is hip-center normalized.
#   Mid-hip = average of left hip (kp 23) and right hip (kp 24)
#   All 543 keypoints are shifted so mid-hip is at origin (0,0,0)
#   This removes signer-position bias so the avatar always
#   appears centered regardless of where the signer stood.
#
# KEYPOINT ORDER VERIFIED:
#   How2Sign Holistic .npy files use MediaPipe Holistic order:
#     0–467  : face (468 points)
#     468–500: pose/body (33 points)
#     501–521: left hand (21 points)
#     522–542: right hand (21 points)
#   The existing norm_pose slices [:, :75, :] which takes the
#   first 75 FACE points — this is WRONG and is also fixed here.
#   The correct avatar slice is [:, 468:543, :] giving
#   33 body + 21 left hand + 21 right hand = 75 points.
#   This fix is applied in normalize_pose_slice() below and
#   must also be applied in Steps 4, 5, and 6 (details noted).
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

# STEP 3.0 — ASL Grammar Transformer (exact copy from Steps 2/4/5)
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
class ASLGrammarTransformerV2:
    ARTICLES     = {"a","an","the"}
    AUXILIARIES  = {"is","are","was","were","am","be","been","being",
                    "do","does","did","has","have","had","will","would",
                    "could","should","shall","may","might","must","can","ought"}
    PREPOSITIONS = {"to","of","in","at","on","for","with","by","from",
                    "into","onto","upon","about","over","under","through",
                    "between","among","within","without","against","toward",
                    "towards","along","across","behind","beside","beyond",
                    "during","except","inside","outside","since","than",
                    "until","unto","up","via","worth"}
    WH_WORDS     = {"what","where","when","who","why","how","which","whom","whose"}
    TIME_WORDS   = {"yesterday","today","tomorrow","now","later","before","after",
                    "soon","recently","always","never","sometimes","already","still",
                    "yet","last","next","every","daily","weekly","monthly","morning",
                    "afternoon","evening","night","monday","tuesday","wednesday",
                    "thursday","friday","saturday","sunday","ago","since"}
    CONTRACTIONS = {
        "don't":"do not","doesn't":"does not","didn't":"did not",
        "won't":"will not","can't":"cannot","isn't":"is not",
        "aren't":"are not","wasn't":"was not","weren't":"were not",
        "i'm":"i am","i've":"i have","i'll":"i will","i'd":"i would",
        "you're":"you are","you've":"you have","you'll":"you will",
        "he's":"he is","she's":"she is","it's":"it is","we're":"we are",
        "they're":"they are","there's":"there is","that's":"that is",
        "let's":"let us","what's":"what is","who's":"who is",
        "we've":"we have","they've":"they have","they'll":"they will",
        "we'll":"we will","we'd":"we would","you'd":"you would",
    }

    def transform(self, sentence):
        if not sentence or str(sentence).lower() == "nan":
            return ""
        s = str(sentence).strip().lower()
        for k, v in self.CONTRACTIONS.items():
            s = s.replace(k, v)
        tokens = re.findall(r"[a-z]+", s)
        if not tokens:
            return ""
        is_wh   = tokens[0] in self.WH_WORDS
        negated = "not" in tokens
        tokens  = [t for t in tokens if t != "not"]
        tokens  = [t for t in tokens if t not in self.ARTICLES]
        if is_wh:
            rest   = [t for t in tokens[1:] if t not in self.AUXILIARIES]
            tokens = [tokens[0]] + rest
        else:
            tokens = [t for t in tokens if t not in self.AUXILIARIES]
        tokens     = [t for t in tokens if t not in self.PREPOSITIONS]
        time_toks  = [t for t in tokens if t in self.TIME_WORDS]
        other_toks = [t for t in tokens if t not in self.TIME_WORDS]
        tokens     = time_toks + other_toks
        if negated:
            tokens.append("not")
        return " ".join(tokens).upper()

# STEP 3.2 — Core functions for segment-level slicing
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

def compute_word_segments(sentence, npy_total_frames, n_tokens):
    """
    Given that the .npy has npy_total_frames frames covering
    the entire sentence, and there are n_tokens gloss words,
    divide the frame range proportionally so each token gets
    an equal-duration, non-overlapping segment.

    This is the best strategy when word-alignment CSVs are not
    available. It ensures:
      - Word 0 always comes from the start of the sentence
      - Word N-1 always comes from the end of the sentence
      - No word ever shares frames with another word
      - The minimum segment is at least 15 frames (half of
        FRAMES_PER_TOKEN) so interpolation has enough data

    Returns list of (start_frame, end_frame) pairs.
    """
    if n_tokens == 0:
        return []
    if n_tokens == 1:
        return [(0, max(npy_total_frames - 1, 0))]

    # Enforce minimum segment size of 15 frames
    min_seg     = 15
    total_min   = min_seg * n_tokens
    usable      = max(npy_total_frames, total_min)
    seg_size    = usable / n_tokens

    segments = []
    for i in range(n_tokens):
        s = int(round(i * seg_size))
        e = int(round((i + 1) * seg_size)) - 1
        # Clamp to actual file length
        s = min(s, npy_total_frames - 1)
        e = min(e, npy_total_frames - 1)
        # Guarantee at least 1 frame
        if e <= s:
            e = min(s + min_seg - 1, npy_total_frames - 1)
        segments.append((s, e))
    return segments


def normalize_pose_slice(arr_slice):
    """
    FIXED keypoint extraction + spatial normalization.

    OLD (wrong): arr[:, :75, :]  → takes 75 face landmarks
    NEW (fixed): arr[:, 468:543, :] → takes body(33) + hands(42)

    MediaPipe Holistic layout in How2Sign .npy files:
      0   – 467 : face (468 landmarks)
      468 – 500 : pose/body (33 landmarks)
      501 – 521 : left hand (21 landmarks)
      522 – 542 : right hand (21 landmarks)
    Total = 543 landmarks (confirmed by your Cell 3 output)

    After slicing [468:543] we get 75 points in this order:
      0  – 32 : body  (matches avatar_points layout in Steps 4/5)
      33 – 53 : left hand
      54 – 74 : right hand

    Hip-center spatial normalization:
      In the 75-point slice:
        body point 23 = left hip  (index 23 in the 75-pt array)
        body point 24 = right hip (index 24 in the 75-pt array)
      mid_hip = (kp[23] + kp[24]) / 2  per frame
      Subtract mid_hip from all 75 keypoints per frame
      This makes the avatar always centered regardless of where
      the signer stood in the original recording.
    """
    # Extract correct keypoint range: body + hands only
    # Shape: (T, 75, 3)
    arr = arr_slice[:, 468:543, :].astype(np.float32)
    T   = arr.shape[0]

    # Hip-center normalization per frame
    # Left hip = body index 23, right hip = body index 24
    left_hip  = arr[:, 23, :]   # shape (T, 3)
    right_hip = arr[:, 24, :]   # shape (T, 3)
    mid_hip   = (left_hip + right_hip) / 2.0  # shape (T, 3)

    # Subtract mid_hip from all 75 keypoints
    # mid_hip[:, np.newaxis, :] broadcasts to (T, 75, 3)
    arr = arr - mid_hip[:, np.newaxis, :]

    # Scale normalization: divide by shoulder width so that
    # the avatar has consistent proportions across different
    # signers. Shoulder width = distance between left(11)
    # and right(12) shoulder landmarks.
    left_shoulder  = arr[:, 11, :2]  # (T, 2)
    right_shoulder = arr[:, 12, :2]  # (T, 2)
    shoulder_width = np.linalg.norm(
        left_shoulder - right_shoulder, axis=1
    )  # (T,)
    scale = shoulder_width.mean() + 1e-6

    arr = arr / scale

    return arr  # shape (T, 75, 3)


def interpolate_to_n_frames(arr, n):
    """
    Resample arr from T frames to exactly n frames using
    linear interpolation (np.linspace index mapping).
    arr shape: (T, 75, 3)
    Returns shape: (n, 75, 3)
    """
    T = arr.shape[0]
    if T == n:
        return arr
    indices = np.linspace(0, T - 1, n)
    lo  = np.floor(indices).astype(int)
    hi  = np.ceil(indices).astype(int)
    hi  = np.minimum(hi, T - 1)
    frac = (indices - lo)[:, np.newaxis, np.newaxis]  # (n, 1, 1)
    return arr[lo] * (1 - frac) + arr[hi] * frac


def load_word_clip(npy_path, start_frame, end_frame, n_frames=FRAMES_PER_TOKEN):
    """
    Load exactly the word's segment from the .npy file.

    Steps:
      1. Load full .npy file
      2. Slice [start_frame : end_frame + 1]  ← only the word's frames
      3. Extract correct keypoints (468:543)
      4. Apply hip-center + scale normalization
      5. Interpolate to exactly n_frames
    """
    arr      = np.load(npy_path, allow_pickle=True).astype(np.float32)
    total    = arr.shape[0]

    # Clamp indices to file bounds
    s = max(0, min(start_frame, total - 1))
    e = max(s, min(end_frame, total - 1))

    segment  = arr[s : e + 1]  # shape (seg_len, 543, 3)

    # Safety: if segment collapsed to 0 frames, use whole file
    if segment.shape[0] == 0:
        segment = arr

    normalized = normalize_pose_slice(segment)    # (seg_len, 75, 3)
    resampled  = interpolate_to_n_frames(normalized, n_frames)  # (n_frames, 75, 3)

    return resampled

def find_best_match(token, vocabulary, threshold=0.6):
    """
    Find best matching word in vocabulary.
    vocabulary is word_to_segment (dict of dicts).
    Exact match first, then fuzzy.
    """
    token_upper = token.upper()
    if token_upper in vocabulary:
        return token_upper, 1.0
    best_match = None
    best_score = 0.0
    for word in vocabulary:
        score = SequenceMatcher(None, token_upper, word).ratio()
        if score > best_score:
            best_score = score
            best_match = word
    if best_score >= threshold:
        return best_match, best_score
    return None, 0.0


def build_sign_sequence(gloss_text, vocabulary):
    """
    REQ-14: Maps ASL gloss tokens to CORRECT keypoint segments.

    For each token:
      1. Finds matching entry in segment vocabulary
      2. Loads only the word's specific frame range
      3. Applies correct keypoint extraction (468:543) + normalization
      4. Interpolates to FRAMES_PER_TOKEN frames

    Returns: dict with tokens, loaded clips, stats
    """
    tokens = gloss_text.upper().split()
    result = {
        "gloss"        : gloss_text,
        "tokens"       : tokens,
        "mappings"     : [],
        "found"        : 0,
        "not_found"    : 0,
        "total_src_frames" : 0,
        "total_out_frames" : 0,
    }

    for token in tokens:
        match, score = find_best_match(token, vocabulary)
        if match:
            seg   = vocabulary[match]
            clip  = load_word_clip(
                seg["path"], seg["start"], seg["end"],
                n_frames=FRAMES_PER_TOKEN
            )
            result["mappings"].append({
                "token"       : token,
                "match"       : match,
                "score"       : round(score, 3),
                "path"        : seg["path"],
                "src_start"   : seg["start"],
                "src_end"     : seg["end"],
                "src_frames"  : seg["n_frames"],
                "out_frames"  : clip.shape[0],
                "clip_shape"  : list(clip.shape),
                "sentence"    : seg["sentence"][:60],
            })
            result["found"]            += 1
            result["total_src_frames"] += seg["n_frames"]
            result["total_out_frames"] += clip.shape[0]
        else:
            result["mappings"].append({
                "token"      : token,
                "match"      : None,
                "score"      : 0.0,
                "path"       : None,
                "src_start"  : 0,
                "src_end"    : 0,
                "src_frames" : 0,
                "out_frames" : 0,
                "clip_shape" : [],
                "sentence"   : "",
            })
            result["not_found"] += 1

    result["coverage"] = round(
        result["found"] / max(len(tokens), 1) * 100, 1
    )
    return result

#   frame-to-frame temporal structure.
#   So the trained weights are valid. We only need to fix what

def get_folder_id(folder_name):
    q = f"name='{folder_name}' and mimeType='application/vnd.google-apps.folder'"
    r = drive_service.files().list(q=q).execute()
    return r['files'][0]['id'] if r['files'] else None

def download_file(file_name, dest_path, folder_id):
    dest_file = dest_path / file_name
    if dest_file.exists():
        print(f"  Exists : {file_name} ({dest_file.stat().st_size/1e6:.1f} MB)")
        return True
    q = f"name='{file_name}' and '{folder_id}' in parents"
    results = drive_service.files().list(
        q=q, fields="files(id, name, size)"
    ).execute()
    if not results['files']:
        print(f"  MISSING : {file_name}")
        return False
    file_id  = results['files'][0]['id']
    size_mb  = int(results['files'][0].get('size', 0)) / 1e6
    print(f"  Downloading : {file_name} ({size_mb:.1f} MB)")
    request  = drive_service.files().get_media(fileId=file_id)
    fh       = io.FileIO(str(dest_file), 'wb')
    dl       = MediaIoBaseDownload(fh, request, chunksize=50*1024*1024)
    done     = False
    while not done:
        _, done = dl.next_chunk()
    fh.close()
    print(f"  Done : {file_name}")
    return True

def upload_file(file_path, folder_id):

            media_body=media, fields='id'
        ).execute()
        print(f"  Uploaded : {file_name} ({os.path.getsize(file_path)/1e6:.1f} MB)")

TRANSITION_FRAMES = 10
MAX_FRAMES        = 900

def normalize_pose_v2(arr_slice):
    """
    FIXED: Extract body+hands keypoints using correct indices.

    OLD (broken in Steps 4/5): arr[:, :75, :]   ← face landmarks
    NEW (fixed here):  arr[:, 468:543, :] ← body + hands

    Layout after extraction (75 points):
      0–32  : body/pose
      33–53 : left hand
      54–74 : right hand

    Normalization:
      1. Hip-center: subtract mid-hip (avg of kp[23] + kp[24])
      2. Scale: divide by mean shoulder width (kp[11] to kp[12])
    """
    arr = arr_slice[:, 468:543, :].astype(np.float32)  # (T, 75, 3)
    T   = arr.shape[0]

    # Hip-center normalization
    mid_hip = (arr[:, 23, :] + arr[:, 24, :]) / 2.0   # (T, 3)
    arr     = arr - mid_hip[:, np.newaxis, :]

    # Scale normalization
    shoulder_w = np.linalg.norm(
        arr[:, 11, :2] - arr[:, 12, :2], axis=1
    )  # (T,)
    scale = shoulder_w.mean() + 1e-6
    arr   = arr / scale

    return arr  # (T, 75, 3)


def interpolate_clip(arr, n):
    """Linear interpolation from T frames to n frames."""
    T = arr.shape[0]
    if T == n:
        return arr
    idx  = np.linspace(0, T - 1, n)
    lo   = np.floor(idx).astype(int)
    hi   = np.minimum(np.ceil(idx).astype(int), T - 1)
    frac = (idx - lo)[:, np.newaxis, np.newaxis]
    return arr[lo] * (1 - frac) + arr[hi] * frac


def load_word_clip_v2(path, start, end, n=FRAMES_PER_TOKEN):
    """
    Load a word's specific frame segment from a .npy file.

    path  : full path to the .npy sentence file
    start : first frame of this word's segment
    end   : last frame of this word's segment (inclusive)
    n     : output frame count (default 30)

    Steps:
      1. Load full .npy  (shape: T_full, 543, 3)
      2. Slice [start:end+1]  (shape: seg_len, 543, 3)
      3. Extract 468:543 + normalize  (shape: seg_len, 75, 3)
      4. Interpolate to n frames  (shape: n, 75, 3)
    """
    arr     = np.load(path, allow_pickle=True).astype(np.float32)
    total   = arr.shape[0]
    s       = max(0, min(start, total - 1))
    e       = max(s + 1, min(end + 1, total))
    segment = arr[s:e]

    if segment.shape[0] == 0:
        segment = arr  # fallback: use whole file

    normed    = normalize_pose_v2(segment)
    resampled = interpolate_clip(normed, n)
    return resampled  # (n, 75, 3)


def cosine_transition(clip_a, clip_b, n_frames):
    """
    Smooth cosine interpolation from last frame of clip_a
    to first frame of clip_b over n_frames.
    Uses cosine easing for natural deceleration/acceleration.
    """
    last_a  = clip_a[-1]   # (75, 3)
    first_b = clip_b[0]    # (75, 3)
    t       = np.linspace(0, 1, n_frames)
    alpha   = 0.5 - 0.5 * np.cos(np.pi * t)  # cosine ease
    return np.stack(
        [(1 - a) * last_a + a * first_b for a in alpha],
        axis=0
    ).astype(np.float32)  # (n_frames, 75, 3)


def find_best_match_v2(token, vocabulary, threshold=0.42):
    """Exact match first, then fuzzy match on word_to_segment keys."""
    tok = token.upper()
    if tok in vocabulary:
        return tok, 1.0
    best_match = None
    best_score = 0.0
    for word in vocabulary:
        sc = SequenceMatcher(None, tok, word).ratio()
        if sc > best_score:
            best_score = sc
            best_match = word
    if best_score >= threshold:
        return best_match, best_score
    return None, 0.0


def build_pose_sequence_v2(gloss_text, vocabulary):
    """
    FIXED version of build_pose_sequence.

    OLD: loaded full sentence .npy files → 100s-1800s of frames
         → PoseFormer received compressed paragraphs as words

    NEW: loads only each word's proportional frame segment
         → PoseFormer receives clean 30-frame word motion clips

    Returns:
      combined : np.ndarray (T_total, 75, 3)
                 T_total = N_tokens * 30 + (N_tokens-1) * 10
      bounds   : list of {token, start_frame, end_frame}
                 for the renderer's per-token highlighting
      info     : dict with stats for debugging
    """
    tokens = gloss_text.upper().split()
    clips  = []
    info   = {"tokens": tokens, "clips": []}

    for tok in tokens:
        match, score = find_best_match_v2(tok, vocabulary)
        if match:
            seg  = vocabulary[match]
            clip = load_word_clip_v2(
                seg["path"], seg["start"], seg["end"],
                n=FRAMES_PER_TOKEN
            )
        else:
            # Unknown word: use a neutral pose (zero motion)
            # This is clearly better than using a random sentence
            clip = np.zeros((FRAMES_PER_TOKEN, 75, 3), dtype=np.float32)
            match = f"~unknown:{tok}"

        clips.append(clip)
        info["clips"].append({
            "token"    : tok,
            "match"    : match,
            "score"    : round(score, 3),
            "src_seg"  : vocabulary.get(match, {}).get("n_frames", 0),
            "out_shape": list(clip.shape),
        })

    if not clips:
        return (
            np.zeros((FRAMES_PER_TOKEN, 75, 3), dtype=np.float32),
            [],
            info
        )

    # Build combined sequence with cosine transitions
    sequence = [clips[0]]
    bounds   = [{
        "token"       : tokens[0],
        "startFrame"  : 0,
        "endFrame"    : FRAMES_PER_TOKEN - 1
    }]
    cur = FRAMES_PER_TOKEN

    for i in range(1, len(clips)):
        # Transition from previous word to this word
        trans = cosine_transition(clips[i - 1], clips[i], TRANSITION_FRAMES)
        sequence.append(trans)
        cur += TRANSITION_FRAMES

        # Append this word's clip
        s = cur
        sequence.append(clips[i])
        cur += FRAMES_PER_TOKEN
        bounds.append({
            "token"      : tokens[i],
            "startFrame" : s,
            "endFrame"   : cur - 1
        })

    combined = np.concatenate(sequence, axis=0).astype(np.float32)

    # Clip to MAX_FRAMES
    if combined.shape[0] > MAX_FRAMES:
        combined = combined[:MAX_FRAMES]

    expected = len(tokens) * FRAMES_PER_TOKEN + (len(tokens) - 1) * TRANSITION_FRAMES
    info["total_frames"] = int(combined.shape[0])
    info["expected_frames"] = expected

    return combined, bounds, info

class PositionalEncoding(nn.Module):
    def __init__(self, d_model, max_len=500, dropout=0.1):
        super().__init__()
        self.dropout = nn.Dropout(dropout)
        pe  = torch.zeros(max_len, d_model)
        pos = torch.arange(0, max_len).unsqueeze(1).float()
        div = torch.exp(
            torch.arange(0, d_model, 2).float() *
            -(np.log(10000.0) / d_model)
        )
        pe[:, 0::2] = torch.sin(pos * div)
        pe[:, 1::2] = torch.cos(pos * div)
        self.register_buffer("pe", pe.unsqueeze(0))

    def forward(self, x):
        return self.dropout(x + self.pe[:, :x.size(1)])

class PoseFormer(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        id_ = cfg["avatar_points"] * cfg["coords"]
        hd  = cfg["hidden_dim"]
        self.input_proj = nn.Sequential(
            nn.Linear(id_, hd), nn.LayerNorm(hd)
        )
        self.pos_enc = PositionalEncoding(
            hd, max_len=cfg["max_seq_len"], dropout=cfg["dropout"]
        )
        enc = nn.TransformerEncoderLayer(
            d_model=hd, nhead=cfg["num_heads"],
            dim_feedforward=hd * 4, dropout=cfg["dropout"],
            batch_first=True, norm_first=True
        )
        self.transformer = nn.TransformerEncoder(
            enc, num_layers=cfg["num_layers"]
        )
        self.output_proj = nn.Linear(hd, id_)
        self.smooth = nn.Sequential(
            nn.Conv1d(id_, id_, kernel_size=5, padding=2, groups=3),
            nn.GELU(),
            nn.Conv1d(id_, id_, kernel_size=3, padding=1, groups=3),
        )

    def make_causal_mask(self, T, dev):
        return torch.triu(
            torch.ones(T, T, device=dev), diagonal=1
        ).bool()

    def forward(self, x):
        B, T, K, C = x.shape
        xf = x.reshape(B, T, K * C)
        h  = self.input_proj(xf)
        h  = self.pos_enc(h)
        h  = self.transformer(h, mask=self.make_causal_mask(T, x.device))
        o  = self.output_proj(h)
        o  = self.smooth(o.permute(0, 2, 1)).permute(0, 2, 1)
        return o.reshape(B, T, K, C)

def get_hand_bones(offset):
    bones = []
    for fs in [1, 5, 9, 13, 17]:
        bones.append((offset, offset + fs))
        for j in range(fs, fs + 3):
            bones.append((offset + j, offset + j + 1))
    return bones

class BoneTransformer(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        id_  = cfg["input_dim"]
        hd   = cfg["hidden_dim"]
        od   = cfg["n_bones"] * 2
        self.pose_encoder = nn.Sequential(
            nn.Linear(id_, hd), nn.LayerNorm(hd),
            nn.GELU(), nn.Linear(hd, hd), nn.LayerNorm(hd),
        )
        self.pos_enc = PositionalEncoding(
            hd, max_len=cfg["max_frames"], dropout=cfg["dropout"]
        )
        enc = nn.TransformerEncoderLayer(
            d_model=hd, nhead=cfg["num_heads"],
            dim_feedforward=hd * 4, dropout=cfg["dropout"],
            batch_first=True, norm_first=True
        )
        self.transformer = nn.TransformerEncoder(
            enc, num_layers=cfg["num_layers"]
        )
        self.bone_decoder = nn.Sequential(
            nn.Linear(hd, hd), nn.GELU(), nn.Linear(hd, od)
        )
        self.temporal_smooth = nn.Sequential(
            nn.Conv1d(od, od, kernel_size=7, padding=3, groups=od),
            nn.GELU(),
            nn.Conv1d(od, od, kernel_size=5, padding=2, groups=od),
            nn.GELU(),
            nn.Conv1d(od, od, kernel_size=3, padding=1, groups=od),
        )

    def make_causal_mask(self, T, dev):
        return torch.triu(
            torch.ones(T, T, device=dev), diagonal=1
        ).bool()

    def forward(self, x):
        B, T, D = x.shape
        h = self.pose_encoder(x)
        h = self.pos_enc(h)
        h = self.transformer(h, mask=self.make_causal_mask(T, x.device))
        r = self.bone_decoder(h)
        r = self.temporal_smooth(r.permute(0, 2, 1)).permute(0, 2, 1)
        return r

def generate_3d_keypoints_v2(gloss_text):
    """
    Full pipeline for one gloss string:
      1. build_pose_sequence_v2 → correct word segments
      2. PoseFormer → refine/smooth the sequence
      3. Export full (T, 75, 3) with x, y, z all kept
      4. Also compute bone vectors for the renderer

    The Z axis is kept (not stripped) so Step 6 can use depth
    for line thickness (closer fingers = thicker lines).
    """
    raw_pose, bounds, info = build_pose_sequence_v2(
        gloss_text, word_to_segment
    )

    # PoseFormer refinement
    pose_t = torch.tensor(
        raw_pose, dtype=torch.float32
    ).unsqueeze(0).to(DEVICE)

    with torch.no_grad():
        with torch.cuda.amp.autocast(enabled=torch.cuda.is_available()):
            refined = poseformer(pose_t)  # (1, T, 75, 3)

    refined_np = refined.squeeze(0).cpu().numpy()  # (T, 75, 3)

    # BoneTransformer — skeletal stability layer
    pose_xy   = refined.squeeze(0)[:, :, :2]        # (T, 75, 2)
    pose_flat = pose_xy.reshape(
        pose_xy.shape[0], -1
    ).unsqueeze(0)                                    # (1, T, 150)

    with torch.no_grad():
        with torch.cuda.amp.autocast(enabled=torch.cuda.is_available()):
            bones_out = model_bone(pose_flat)          # (1, T, 104)

    bones_np = bones_out.squeeze(0).cpu().numpy()     # (T, n_bones*2)

class ASLGrammarTransformerV2Bridge:
    """Minimal copy for cache-building in this cell."""
    ARTICLES={" a","an","the"}
    AUXILIARIES={"is","are","was","were","am","be","been","being",
                 "do","does","did","has","have","had","will","would",
                 "could","should","shall","may","might","must","can","ought"}
    PREPOSITIONS={"to","of","in","at","on","for","with","by","from",
                  "into","onto","upon","about","over","under","through",
                  "between","among","within","without","against","toward",
                  "towards","along","across","behind","beside","beyond",
                  "during","except","inside","outside","since","than",
                  "until","unto","up","via","worth"}
    WH_WORDS={"what","where","when","who","why","how","which","whom","whose"}
    TIME_WORDS={"yesterday","today","tomorrow","now","later","before","after",
                "soon","recently","always","never","sometimes","already","still",
                "yet","last","next","every","daily","weekly","monthly","morning",
                "afternoon","evening","night","monday","tuesday","wednesday",
                "thursday","friday","saturday","sunday","ago","since"}
    CONTRACTIONS={
        "don't":"do not","doesn't":"does not","didn't":"did not",
        "won't":"will not","can't":"cannot","isn't":"is not",
        "aren't":"are not","wasn't":"was not","weren't":"were not",
        "i'm":"i am","i've":"i have","i'll":"i will","i'd":"i would",
        "you're":"you are","you've":"you have","you'll":"you will",
        "he's":"he is","she's":"she is","it's":"it is","we're":"we are",
        "they're":"they are","there's":"there is","that's":"that is",
        "let's":"let us","what's":"what is","who's":"who is",
        "we've":"we have","they've":"they have","they'll":"they will",
        "we'll":"we will","we'd":"we would","you'd":"you would",
    }
    def transform(self, sentence):
        if not sentence: return ""
        s=str(sentence).strip().lower()
        for k,v in self.CONTRACTIONS.items(): s=s.replace(k,v)
        t=_re.findall(r"[a-z]+",s)
        if not t: return ""
        is_wh=t[0] in self.WH_WORDS
        neg="not" in t
        t=[x for x in t if x!="not"]
        t=[x for x in t if x not in self.ARTICLES]
        if is_wh:
            r=[x for x in t[1:] if x not in self.AUXILIARIES]
            t=[t[0]]+r
        else: t=[x for x in t if x not in self.AUXILIARIES]
        t=[x for x in t if x not in self.PREPOSITIONS]
        t=[*[x for x in t if x in self.TIME_WORDS],
           *[x for x in t if x not in self.TIME_WORDS]]
        if neg: t.append("not")
        return " ".join(t).upper()

def normalize_pose(arr):
    arr    = arr[:, :CFG_POSE["avatar_points"], :].astype(np.float32)
    center = arr[:, 0:1, :2].mean(axis=0, keepdims=True)
    scale  = np.std(arr[:, :33, :2]) + 1e-6
    result = arr.copy()
    result[:, :, :2] = (arr[:, :, :2] - center) / scale
    result[:, :, 2]  = arr[:, :, 2] / scale
    return result


def load_pose_clip(npy_path, n_frames):
    arr   = np.load(npy_path, allow_pickle=True).astype(np.float32)
    arr   = normalize_pose(arr)
    total = arr.shape[0]
    if total == n_frames:
        return arr
    elif total > n_frames:
        idx = np.linspace(0, total-1, n_frames, dtype=int)
        return arr[idx]
    else:
        pad = np.tile(arr[-1:], (n_frames-total, 1, 1))
        return np.concatenate([arr, pad], axis=0)


def make_transition(pose_a, pose_b, n_frames):
    last_a = pose_a[-1]
    first_b = pose_b[0]
    alphas  = np.linspace(0, 1, n_frames)
    return np.stack(
        [(1-a)*last_a + a*first_b for a in alphas], axis=0
    )


def build_pose_sequence(gloss_text, vocabulary):
    tokens = gloss_text.upper().split()
    clips  = []

    for token in tokens:
        npy_path = vocabulary.get(token)
        if not npy_path:
            best_match, best_score = None, 0.0
            for word in vocabulary:
                score = SequenceMatcher(None, token, word).ratio()
                if score > best_score:
                    best_score = score
                    best_match = word
            if best_score >= 0.6:
                npy_path = vocabulary[best_match]
        if npy_path:
            clips.append(
                load_pose_clip(npy_path, CFG_POSE["frames_per_token"])
            )

    if not clips:
        return np.zeros(
            (CFG_POSE["frames_per_token"],
             CFG_POSE["avatar_points"], 3),
            dtype=np.float32
        )

    sequence = [clips[0]]
    for i in range(1, len(clips)):
        sequence.append(
            make_transition(
                clips[i-1], clips[i],
                CFG_POSE["transition_frames"]
            )
        )
        sequence.append(clips[i])

    combined = np.concatenate(sequence, axis=0)
    if combined.shape[0] > CFG_POSE["max_seq_len"]:
        combined = combined[:CFG_POSE["max_seq_len"]]
    return combined.astype(np.float32)

class PositionalEncoding(nn.Module):
    def __init__(self, d_model, max_len=500, dropout=0.1):
        super().__init__()
        self.dropout = nn.Dropout(dropout)
        pe  = torch.zeros(max_len, d_model)
        pos = torch.arange(0, max_len).unsqueeze(1).float()
        div = torch.exp(
            torch.arange(0, d_model, 2).float() *
            -(np.log(10000.0) / d_model)
        )
        pe[:, 0::2] = torch.sin(pos * div)
        pe[:, 1::2] = torch.cos(pos * div)
        self.register_buffer("pe", pe.unsqueeze(0))

    def forward(self, x):
        return self.dropout(x + self.pe[:, :x.size(1)])


class PoseFormer(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        input_dim  = cfg["avatar_points"] * cfg["coords"]
        hidden_dim = cfg["hidden_dim"]
        self.input_proj = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
        )
        self.pos_enc = PositionalEncoding(
            hidden_dim,
            max_len=cfg["max_seq_len"],
            dropout=cfg["dropout"]
        )
        enc_layer = nn.TransformerEncoderLayer(
            d_model=hidden_dim,
            nhead=cfg["num_heads"],
            dim_feedforward=hidden_dim * 4,
            dropout=cfg["dropout"],
            batch_first=True,
            norm_first=True,
        )
        self.transformer = nn.TransformerEncoder(
            enc_layer,
            num_layers=cfg["num_layers"]
        )
        self.output_proj = nn.Linear(hidden_dim, input_dim)
        self.smooth = nn.Sequential(
            nn.Conv1d(input_dim, input_dim,
                      kernel_size=5, padding=2, groups=3),
            nn.GELU(),
            nn.Conv1d(input_dim, input_dim,
                      kernel_size=3, padding=1, groups=3),
        )

    def make_causal_mask(self, T, device):
        return torch.triu(
            torch.ones(T, T, device=device), diagonal=1
        ).bool()

    def forward(self, x):
        B, T, K, C = x.shape
        x_flat = x.reshape(B, T, K * C)
        h      = self.input_proj(x_flat)
        h      = self.pos_enc(h)
        mask   = self.make_causal_mask(T, x.device)
        h      = self.transformer(h, mask=mask)
        out    = self.output_proj(h)
        out_t  = out.permute(0, 2, 1)
        out_s  = self.smooth(out_t)
        out    = out_s.permute(0, 2, 1)
        return out.reshape(B, T, K, C)

class PoseDataset(Dataset):
    def __init__(self, pairs, vocabulary):
        self.vocabulary = vocabulary
        self.valid      = []
        for p in tqdm(pairs, desc="  Indexing", leave=False):
            gloss = p.get("gloss", "").strip()
            if not gloss:
                continue
            tokens = gloss.upper().split()
            if any(t in vocabulary for t in tokens):
                self.valid.append(p)
        print(f"  Valid pairs : {len(self.valid):,}")

    def __len__(self):
        return len(self.valid)

    def __getitem__(self, idx):
        pair  = self.valid[idx]
        gloss = pair["gloss"].upper()
        pose  = torch.tensor(
            build_pose_sequence(gloss, self.vocabulary),
            dtype=torch.float32
        )
        T = pose.shape[0]
        if T < 2:
            pose = pose.repeat(2, 1, 1)
        return {
            "input" : pose[:-1],
            "target": pose[1:],
            "gloss" : gloss,
        }


def collate_fn(batch):
    max_t   = max(b["input"].shape[0] for b in batch)
    K       = batch[0]["input"].shape[1]
    C       = batch[0]["input"].shape[2]
    inputs  = torch.zeros(len(batch), max_t, K, C)
    targets = torch.zeros(len(batch), max_t, K, C)
    for i, b in enumerate(batch):
        t = b["input"].shape[0]
        inputs[i,  :t] = b["input"]
        targets[i, :t] = b["target"]
    return {
        "input" : inputs,
        "target": targets,
        "gloss" : [b["gloss"] for b in batch],
    }

def run_epoch(model, loader, optimizer, scaler, device, train=True):
    model.train() if train else model.eval()
    total, steps = 0.0, 0
    if train:
        optimizer.zero_grad()
    ctx  = torch.enable_grad() if train else torch.no_grad()
    pbar = tqdm(
        loader,
        desc="  Train" if train else "  Val  ",
        leave=False
    )
    with ctx:
        for idx, batch in enumerate(pbar):
            inp = batch["input"].to(device)
            tgt = batch["target"].to(device)
            with torch.cuda.amp.autocast(
                enabled=torch.cuda.is_available()
            ):
                out  = model(inp)
                loss = criterion(out, tgt)
                if train:
                    loss = loss / ACCUM
            if train:
                scaler.scale(loss).backward()
                if (idx + 1) % ACCUM == 0:
                    scaler.unscale_(optimizer)
                    torch.nn.utils.clip_grad_norm_(
                        model.parameters(),
                        CFG_POSE["grad_clip"]
                    )
                    scaler.step(optimizer)
                    scaler.update()
                    optimizer.zero_grad()
                total += loss.item() * ACCUM
            else:
                total += loss.item()
            steps += 1
            pbar.set_postfix(loss=f"{loss.item():.5f}")
    return total / max(steps, 1)

class PositionalEncoding(nn.Module):
    def __init__(self, d_model, max_len=500, dropout=0.1):
        super().__init__()
        self.dropout = nn.Dropout(dropout)
        pe  = torch.zeros(max_len, d_model)
        pos = torch.arange(0, max_len).unsqueeze(1).float()
        div = torch.exp(
            torch.arange(0, d_model, 2).float() *
            -(np.log(10000.0) / d_model)
        )
        pe[:, 0::2] = torch.sin(pos * div)
        pe[:, 1::2] = torch.cos(pos * div)
        self.register_buffer("pe", pe.unsqueeze(0))

    def forward(self, x):
        return self.dropout(x + self.pe[:, :x.size(1)])

class PoseFormer(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        input_dim  = cfg["avatar_points"] * cfg["coords"]
        hidden_dim = cfg["hidden_dim"]
        self.input_proj = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
        )
        self.pos_enc = PositionalEncoding(
            hidden_dim, max_len=cfg["max_seq_len"],
            dropout=cfg["dropout"]
        )
        enc_layer = nn.TransformerEncoderLayer(
            d_model=hidden_dim, nhead=cfg["num_heads"],
            dim_feedforward=hidden_dim*4,
            dropout=cfg["dropout"],
            batch_first=True, norm_first=True,
        )
        self.transformer = nn.TransformerEncoder(
            enc_layer, num_layers=cfg["num_layers"]
        )
        self.output_proj = nn.Linear(hidden_dim, input_dim)
        self.smooth = nn.Sequential(
            nn.Conv1d(input_dim, input_dim,
                      kernel_size=5, padding=2, groups=3),
            nn.GELU(),
            nn.Conv1d(input_dim, input_dim,
                      kernel_size=3, padding=1, groups=3),
        )

    def make_causal_mask(self, T, device):
        return torch.triu(
            torch.ones(T, T, device=device), diagonal=1
        ).bool()

    def forward(self, x):
        B, T, K, C = x.shape
        x_flat = x.reshape(B, T, K*C)
        h      = self.input_proj(x_flat)
        h      = self.pos_enc(h)
        mask   = self.make_causal_mask(T, x.device)
        h      = self.transformer(h, mask=mask)
        out    = self.output_proj(h)
        out_t  = out.permute(0, 2, 1)
        out_s  = self.smooth(out_t)
        out    = out_s.permute(0, 2, 1)
        return out.reshape(B, T, K, C)

# ── Pose functions ────────────────────────────────────────────────
def normalize_pose(arr):
    arr    = arr[:, :75, :].astype(np.float32)
    center = arr[:, 0:1, :2].mean(axis=0, keepdims=True)
    scale  = np.std(arr[:, :33, :2]) + 1e-6
    result = arr.copy()
    result[:, :, :2] = (arr[:, :, :2] - center) / scale
    result[:, :, 2]  = arr[:, :, 2] / scale
    return result

def load_pose_clip(npy_path, n_frames):
    arr   = np.load(npy_path, allow_pickle=True).astype(np.float32)
    arr   = normalize_pose(arr)
    total = arr.shape[0]
    if total == n_frames:
        return arr
    elif total > n_frames:
        idx = np.linspace(0, total-1, n_frames, dtype=int)
        return arr[idx]
    else:
        pad = np.tile(arr[-1:], (n_frames-total, 1, 1))
        return np.concatenate([arr, pad], axis=0)

def make_transition(pose_a, pose_b, n_frames):
    last_a  = pose_a[-1]
    first_b = pose_b[0]
    alphas  = np.linspace(0, 1, n_frames)
    return np.stack(
        [(1-a)*last_a + a*first_b for a in alphas], axis=0
    )

def build_pose_sequence(gloss_text, vocabulary):
    tokens = gloss_text.upper().split()
    clips  = []
    for token in tokens:
        npy_path = vocabulary.get(token)
        if not npy_path:
            best_match, best_score = None, 0.0
            for word in vocabulary:
                score = SequenceMatcher(
                    None, token, word
                ).ratio()
                if score > best_score:
                    best_score = score
                    best_match = word
            if best_score >= 0.6:
                npy_path = vocabulary[best_match]
        if npy_path:
            clips.append(load_pose_clip(
                npy_path, CFG_BONE["frames_per_token"]
            ))
    if not clips:
        return np.zeros(
            (CFG_BONE["frames_per_token"], 75, 3),
            dtype=np.float32
        )
    sequence = [clips[0]]
    for i in range(1, len(clips)):
        sequence.append(make_transition(
            clips[i-1], clips[i],
            CFG_BONE["transition_frames"]
        ))
        sequence.append(clips[i])
    combined = np.concatenate(sequence, axis=0)
    if combined.shape[0] > CFG_BONE["max_frames"]:
        combined = combined[:CFG_BONE["max_frames"]]
    return combined.astype(np.float32)

def pose_to_bones(pose_seq):
    frames = pose_seq.shape[0]
    bones  = np.zeros(
        (frames, len(ALL_BONES), 2),
        dtype=np.float32
    )
    for i, (p, c) in enumerate(ALL_BONES):
        bones[:, i, :] = (
            pose_seq[:, c, :2] - pose_seq[:, p, :2]
        )
    return bones

# ── Rebuild BoneTransformer ───────────────────────────────────────
class BoneTransformer(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        input_dim  = cfg["input_dim"]
        hidden_dim = cfg["hidden_dim"]
        output_dim = cfg["n_bones"] * 2
        self.pose_encoder = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
            nn.GELU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
        )
        self.pos_enc = PositionalEncoding(
            hidden_dim, max_len=cfg["max_frames"],
            dropout=cfg["dropout"]
        )
        enc_layer = nn.TransformerEncoderLayer(
            d_model=hidden_dim, nhead=cfg["num_heads"],
            dim_feedforward=hidden_dim*4,
            dropout=cfg["dropout"],
            batch_first=True, norm_first=True,
        )
        self.transformer = nn.TransformerEncoder(
            enc_layer, num_layers=cfg["num_layers"]
        )
        self.bone_decoder = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.GELU(),
            nn.Linear(hidden_dim, output_dim),
        )
        self.temporal_smooth = nn.Sequential(
            nn.Conv1d(output_dim, output_dim,
                      kernel_size=7, padding=3,
                      groups=output_dim),
            nn.GELU(),
            nn.Conv1d(output_dim, output_dim,
                      kernel_size=5, padding=2,
                      groups=output_dim),
            nn.GELU(),
            nn.Conv1d(output_dim, output_dim,
                      kernel_size=3, padding=1,
                      groups=output_dim),
        )

    def make_causal_mask(self, T, device):
        return torch.triu(
            torch.ones(T, T, device=device),
            diagonal=1
        ).bool()

    def forward(self, x):
        B, T, D   = x.shape
        h         = self.pose_encoder(x)
        h         = self.pos_enc(h)
        mask      = self.make_causal_mask(T, x.device)
        h         = self.transformer(h, mask=mask)
        bones_raw = self.bone_decoder(h)
        bones_t   = bones_raw.permute(0, 2, 1)
        bones_s   = self.temporal_smooth(bones_t)
        bones_out = bones_s.permute(0, 2, 1)
        return bones_out

class BoneDataset(Dataset):
    def __init__(self, pairs, vocabulary,
                 poseformer, device):
        self.vocabulary = vocabulary
        self.poseformer = poseformer
        self.device     = device
        self.valid      = []
        for p in tqdm(
            pairs, desc="  Indexing", leave=False
        ):
            gloss = p.get("gloss", "").strip()
            if not gloss:
                continue
            tokens = gloss.upper().split()
            if any(t in vocabulary for t in tokens):
                self.valid.append(p)
        print(f"  Valid pairs : {len(self.valid):,}")

    def __len__(self):
        return len(self.valid)

    def __getitem__(self, idx):
        pair     = self.valid[idx]
        gloss    = pair["gloss"].upper()
        raw_pose = build_pose_sequence(
            gloss, self.vocabulary
        )
        pose_t = torch.tensor(
            raw_pose, dtype=torch.float32
        ).unsqueeze(0).to(self.device)
        with torch.no_grad():
            refined = self.poseformer(pose_t)
        refined_np = refined.squeeze(0).cpu().numpy()
        bones      = pose_to_bones(refined_np)
        pose_xy = torch.tensor(
            refined_np[:, :, :2].reshape(
                refined_np.shape[0], -1
            ),
            dtype=torch.float32
        )
        bone_tgt = torch.tensor(
            bones.reshape(bones.shape[0], -1),
            dtype=torch.float32
        )
        return {
            "pose_xy" : pose_xy,
            "bone_tgt": bone_tgt,
            "gloss"   : gloss,
        }


def collate_bone(batch):
    max_t    = max(b["pose_xy"].shape[0] for b in batch)
    pose_dim = batch[0]["pose_xy"].shape[1]
    bone_dim = batch[0]["bone_tgt"].shape[1]
    poses    = torch.zeros(len(batch), max_t, pose_dim)
    bones    = torch.zeros(len(batch), max_t, bone_dim)
    for i, b in enumerate(batch):
        t = b["pose_xy"].shape[0]
        poses[i, :t] = b["pose_xy"]
        bones[i, :t] = b["bone_tgt"]
    return {
        "pose_xy" : poses,
        "bone_tgt": bones,
        "gloss"   : [b["gloss"] for b in batch],
    }

def run_epoch(model, loader, optimizer,
              scaler, device, train=True):
    model.train() if train else model.eval()
    total, steps = 0.0, 0
    if train:
        optimizer.zero_grad()
    ctx  = (torch.enable_grad()
            if train else torch.no_grad())
    pbar = tqdm(
        loader,
        desc="  Train" if train else "  Val  ",
        leave=False
    )
    with ctx:
        for idx, batch in enumerate(pbar):
            pose = batch["pose_xy"].to(device)
            bone = batch["bone_tgt"].to(device)
            with torch.cuda.amp.autocast(
                enabled=torch.cuda.is_available()
            ):
                out  = model(pose)
                loss = criterion(out, bone)
                if train:
                    loss = loss / ACCUM
            if train:
                scaler.scale(loss).backward()
                if (idx + 1) % ACCUM == 0:
                    scaler.unscale_(optimizer)
                    torch.nn.utils.clip_grad_norm_(
                        model.parameters(),
                        CFG_BONE["grad_clip"]
                    )
                    scaler.step(optimizer)
                    scaler.update()
                    optimizer.zero_grad()
                total += loss.item() * ACCUM
            else:
                total += loss.item()
            steps += 1
            pbar.set_postfix(
                loss=f"{loss.item():.5f}"
            )
    return total / max(steps, 1)

# ── Rebuild Model Architectures ───────────────────────────────────
class PositionalEncoding(nn.Module):
    def __init__(self, d_model, max_len=500, dropout=0.1):
        super().__init__()
        self.dropout = nn.Dropout(dropout)
        pe  = torch.zeros(max_len, d_model)
        pos = torch.arange(0, max_len).unsqueeze(1).float()
        div = torch.exp(
            torch.arange(0, d_model, 2).float() *
            -(np.log(10000.0) / d_model)
        )
        pe[:, 0::2] = torch.sin(pos * div)
        pe[:, 1::2] = torch.cos(pos * div)
        self.register_buffer("pe", pe.unsqueeze(0))

    def forward(self, x):
        return self.dropout(x + self.pe[:, :x.size(1)])


class PoseFormer(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        input_dim  = cfg["avatar_points"] * cfg["coords"]
        hidden_dim = cfg["hidden_dim"]
        self.input_proj = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
        )
        self.pos_enc = PositionalEncoding(
            hidden_dim, max_len=cfg["max_seq_len"],
            dropout=cfg["dropout"]
        )
        enc_layer = nn.TransformerEncoderLayer(
            d_model=hidden_dim, nhead=cfg["num_heads"],
            dim_feedforward=hidden_dim * 4,
            dropout=cfg["dropout"],
            batch_first=True, norm_first=True,
        )
        self.transformer = nn.TransformerEncoder(
            enc_layer, num_layers=cfg["num_layers"]
        )
        self.output_proj = nn.Linear(hidden_dim, input_dim)
        self.smooth = nn.Sequential(
            nn.Conv1d(input_dim, input_dim,
                      kernel_size=5, padding=2, groups=3),
            nn.GELU(),
            nn.Conv1d(input_dim, input_dim,
                      kernel_size=3, padding=1, groups=3),
        )

    def make_causal_mask(self, T, device):
        return torch.triu(
            torch.ones(T, T, device=device), diagonal=1
        ).bool()

    def forward(self, x):
        B, T, K, C = x.shape
        x_flat = x.reshape(B, T, K * C)
        h      = self.input_proj(x_flat)
        h      = self.pos_enc(h)
        mask   = self.make_causal_mask(T, x.device)
        h      = self.transformer(h, mask=mask)
        out    = self.output_proj(h)
        out_t  = out.permute(0, 2, 1)
        out_s  = self.smooth(out_t)
        out    = out_s.permute(0, 2, 1)
        return out.reshape(B, T, K, C)


class BoneTransformer(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        input_dim  = cfg["input_dim"]
        hidden_dim = cfg["hidden_dim"]
        output_dim = cfg["n_bones"] * 2
        self.pose_encoder = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
            nn.GELU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
        )
        self.pos_enc = PositionalEncoding(
            hidden_dim, max_len=cfg["max_frames"],
            dropout=cfg["dropout"]
        )
        enc_layer = nn.TransformerEncoderLayer(
            d_model=hidden_dim, nhead=cfg["num_heads"],
            dim_feedforward=hidden_dim * 4,
            dropout=cfg["dropout"],
            batch_first=True, norm_first=True,
        )
        self.transformer = nn.TransformerEncoder(
            enc_layer, num_layers=cfg["num_layers"]
        )
        self.bone_decoder = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.GELU(),
            nn.Linear(hidden_dim, output_dim),
        )
        self.temporal_smooth = nn.Sequential(
            nn.Conv1d(output_dim, output_dim,
                      kernel_size=7, padding=3, groups=output_dim),
            nn.GELU(),
            nn.Conv1d(output_dim, output_dim,
                      kernel_size=5, padding=2, groups=output_dim),
            nn.GELU(),
            nn.Conv1d(output_dim, output_dim,
                      kernel_size=3, padding=1, groups=output_dim),
        )

    def make_causal_mask(self, T, device):
        return torch.triu(
            torch.ones(T, T, device=device), diagonal=1
        ).bool()

    def forward(self, x):
        B, T, D   = x.shape
        h         = self.pose_encoder(x)
        h         = self.pos_enc(h)
        mask      = self.make_causal_mask(T, x.device)
        h         = self.transformer(h, mask=mask)
        bones_raw = self.bone_decoder(h)
        bones_t   = bones_raw.permute(0, 2, 1)
        bones_s   = self.temporal_smooth(bones_t)
        bones_out = bones_s.permute(0, 2, 1)
        return bones_out

# ── Rebuild Pose Utility Functions ────────────────────────────────
def normalize_pose(arr):
    arr    = arr[:, :poseformer_cfg["avatar_points"], :].astype(np.float32)
    center = arr[:, 0:1, :2].mean(axis=0, keepdims=True)
    scale  = np.std(arr[:, :33, :2]) + 1e-6
    result = arr.copy()
    result[:, :, :2] = (arr[:, :, :2] - center) / scale
    result[:, :, 2]  =  arr[:, :, 2]  / scale
    return result

def load_pose_clip(npy_path, n_frames):
    arr   = np.load(npy_path, allow_pickle=True).astype(np.float32)
    arr   = normalize_pose(arr)
    total = arr.shape[0]
    if total == n_frames:
        return arr
    elif total > n_frames:
        idx = np.linspace(0, total - 1, n_frames, dtype=int)
        return arr[idx]
    else:
        pad = np.tile(arr[-1:], (n_frames - total, 1, 1))
        return np.concatenate([arr, pad], axis=0)

def make_transition(pose_a, pose_b, n_frames):
    alphas = np.linspace(0, 1, n_frames)
    return np.stack(
        [(1 - a) * pose_a[-1] + a * pose_b[0] for a in alphas],
        axis=0
    )

def build_pose_sequence(gloss_text, vocabulary):
    tokens = gloss_text.upper().split()
    clips  = []
    FPT    = CFG_BONE["frames_per_token"]
    TF     = CFG_BONE["transition_frames"]
    MF     = CFG_BONE["max_frames"]
    for token in tokens:
        npy_path = vocabulary.get(token)
        if not npy_path:
            best_match, best_score = None, 0.0
            for word in vocabulary:
                score = SequenceMatcher(None, token, word).ratio()
                if score > best_score:
                    best_score = score
                    best_match = word
            if best_score >= 0.6:
                npy_path = vocabulary[best_match]
        if npy_path:
            clips.append(load_pose_clip(npy_path, FPT))
    if not clips:
        return np.zeros((FPT, poseformer_cfg["avatar_points"], 3),
                        dtype=np.float32)
    sequence = [clips[0]]
    for i in range(1, len(clips)):
        sequence.append(make_transition(clips[i-1], clips[i], TF))
        sequence.append(clips[i])
    combined = np.concatenate(sequence, axis=0)
    if combined.shape[0] > MF:
        combined = combined[:MF]
    return combined.astype(np.float32)

def generate_3d_keypoints(gloss_text):
    """
    Exports FULL 3D keypoints (x, y, z) — no Z stripping.

def normalize_pose_v2(arr_slice):
    body  = arr_slice[:, 0:33, :]
    hands = arr_slice[:, 501:543, :]
    arr = np.concatenate((body, hands), axis=1).astype(np.float32)
    
    # Memorize true missing points before scaling
    missing_mask = (np.abs(arr[:, :, 0]) < 1e-5) & (np.abs(arr[:, :, 1]) < 1e-5)
    
    # 🚨 FIX 1: Exact Size Matching
    # Find valid frames to calculate shoulders
    valid_sh = (np.abs(arr[:, 11, :]).sum(axis=1) > 1e-5) & (np.abs(arr[:, 12, :]).sum(axis=1) > 1e-5)
    if valid_sh.sum() > 0:
        mid_sh = (arr[valid_sh, 11, :] + arr[valid_sh, 12, :]) / 2.0
        median_shoulder = np.median(mid_sh, axis=0)
    else:
        median_shoulder = np.array([0.5, 0.5])
        
    # Anchor the skeleton perfectly to the shoulders
    arr = arr - median_shoulder[np.newaxis, np.newaxis, :]
    
    # Measure the width and forcefully shrink/grow to EXACTLY 0.18 (The 15k dataset size)
    shoulder_width = np.linalg.norm(arr[:, 11, :] - arr[:, 12, :], axis=1)
    true_width = np.percentile(shoulder_width[valid_sh], 90) if valid_sh.sum() > 0 else 0.18
    true_width = np.clip(true_width, 0.05, 1.0) # Prevent math explosions
    
    scale_factor = 0.18 / true_width 
    arr = arr * scale_factor
    
    # 🚨 FIX 2: Collapse Detection (The "Hand at Elbow" Fix)
    # If MediaPipe panicked and merged the wrist into the elbow, hide the arm!
    left_arm_len = np.linalg.norm(arr[:, 13, :] - arr[:, 15, :], axis=1)
    arr[left_arm_len < 0.02, 15, :] = 0.0 # Force left wrist to disappear
    arr[left_arm_len < 0.02, 13, :] = 0.0 # Force left elbow to disappear
    
    right_arm_len = np.linalg.norm(arr[:, 14, :] - arr[:, 16, :], axis=1)
    arr[right_arm_len < 0.02, 16, :] = 0.0 # Force right wrist to disappear
    arr[right_arm_len < 0.02, 14, :] = 0.0 # Force right elbow to disappear
    
    # Rigid Torso
    arr[:, 23, 0] = arr[:, 11, 0]        
    arr[:, 23, 1] = arr[:, 11, 1] + 0.35 
    arr[:, 24, 0] = arr[:, 12, 0]        
    arr[:, 24, 1] = arr[:, 12, 1] + 0.35 
    
    # Push missing points back to exactly 0.0
    arr[missing_mask] = 0.0
    
    return arr

def interpolate_clip(arr, n):
    T = arr.shape[0]
    if T == n: return arr
    idx = np.linspace(0, T - 1, n)
    lo = np.floor(idx).astype(int); hi = np.minimum(np.ceil(idx).astype(int), T - 1)
    frac = (idx - lo)[:, np.newaxis, np.newaxis]
    
    p_lo = arr[lo]
    p_hi = arr[hi]
    miss_lo = (np.abs(p_lo[:, :, 0]) < 1e-5) & (np.abs(p_lo[:, :, 1]) < 1e-5)
    miss_hi = (np.abs(p_hi[:, :, 0]) < 1e-5) & (np.abs(p_hi[:, :, 1]) < 1e-5)
    
    interp = p_lo * (1 - frac) + p_hi * frac
    interp[miss_lo | miss_hi] = 0.0
    
    return interp

def load_word_clip_2d(path, start, end, n=FPT):
    arr = np.load(path, allow_pickle=True).astype(np.float32)
    total = arr.shape[0]
    seg = arr[max(0, min(start, total - 1)):max(start + 1, min(end + 1, total))]
    if seg.shape[0] == 0: seg = arr
    
    for hand_start, hand_end in [(501, 522), (522, 543)]:
        wrist_sums = np.abs(seg[:, hand_start, :2]).sum(axis=1)
        valid_frames = np.where(wrist_sums > 1e-5)[0]
        
        if 0 < len(valid_frames) < seg.shape[0]:
            for t in range(seg.shape[0]):
                if wrist_sums[t] <= 1e-5:
                    nearest = valid_frames[np.argmin(np.abs(valid_frames - t))]
                    seg[t, hand_start:hand_end] = seg[nearest, hand_start:hand_end]
                    
    normed = normalize_pose_v2(seg)
    return interpolate_clip(normed, n)[:, :, :2]

function toASL(s) {
  if(!s) return '';
  s = s.trim().toLowerCase().replace("don't","do not").replace("doesn't","does not");
  let t = (s.match(/[a-z]+/g) || []).filter(x => x !== 'not' && !['a','an','the','to','of','in','at','on','for','with','by','from'].includes(x));

  return t.join(' ').toUpperCase();

Root cause analysis of all 3 previous failures:

  v3  (no aug):   95% train / 52% val — 43% gap

# NORMALIZATION  —  THE CRITICAL FIX
# ══════════════════════════════════════════════════════════════════════════════
def normalize_landmarks(x: np.ndarray) -> np.ndarray:
    """
    Wrist-relative normalization for hands; shoulder-relative for pose.

    Input  : x  shape (T, 225) — raw MediaPipe landmark coordinates
    Output : out shape (T, 225) — normalized

    Feature layout:
      Features  0 : 63  = right hand  (landmarks 0-20, each xyz)
      Features 63 :126  = left  hand  (landmarks 0-20, each xyz)
      Features 126:225  = pose        (landmarks 0-32, each xyz)

    Reference landmarks:
      RH landmark  0 = wrist          → features  0: 3
      RH landmark  9 = middle-MCP     → features 27:30   (scale = wrist→MCP dist)
      LH landmark  0 = wrist          → features 63:66
      LH landmark  9 = middle-MCP     → features 90:93
      Pose landmark 11 = left shoulder → features 159:162
      Pose landmark 12 = right shoulder → features 162:165

    Frames where a hand is not detected (all zeros) are left as zeros.
    This function is IDENTICAL to what you must apply in your FastAPI backend.
    """
    out = x.copy().astype(np.float32)

    # ── Right hand ────────────────────────────────────────────────────────────
    rh_wrist = out[:, 0:3].copy()          # (T, 3)
    rh_mcp   = out[:, 27:30].copy()        # middle-finger MCP (T, 3)
    rh_scale = np.linalg.norm(rh_mcp - rh_wrist, axis=1, keepdims=True)  # (T,1)
    rh_vis   = (rh_scale > 1e-4).squeeze(1)   # frames where hand is detected
    if rh_vis.any():
        s_safe = np.where(rh_scale > 1e-4, rh_scale, 1.0)
        for lm in range(21):
            s = lm * 3
            out[rh_vis, s:s+3] = (
                (out[rh_vis, s:s+3] - rh_wrist[rh_vis]) / s_safe[rh_vis]
            )

    # ── Left hand ─────────────────────────────────────────────────────────────
    lh_wrist = out[:, 63:66].copy()
    lh_mcp   = out[:, 90:93].copy()
    lh_scale = np.linalg.norm(lh_mcp - lh_wrist, axis=1, keepdims=True)
    lh_vis   = (lh_scale > 1e-4).squeeze(1)
    if lh_vis.any():
        s_safe = np.where(lh_scale > 1e-4, lh_scale, 1.0)
        for lm in range(21):
            s = 63 + lm * 3
            out[lh_vis, s:s+3] = (
                (out[lh_vis, s:s+3] - lh_wrist[lh_vis]) / s_safe[lh_vis]
            )

    # ── Pose ──────────────────────────────────────────────────────────────────
    ls = out[:, 159:162].copy()   # left  shoulder
    rs = out[:, 162:165].copy()   # right shoulder
    pose_origin = (ls + rs) / 2.0
    pose_scale  = np.linalg.norm(ls - rs, axis=1, keepdims=True)
    p_vis       = (pose_scale > 1e-4).squeeze(1)
    if p_vis.any():
        s_safe = np.where(pose_scale > 1e-4, pose_scale, 1.0)
        for lm in range(33):
            s = 126 + lm * 3
            out[p_vis, s:s+3] = (
                (out[p_vis, s:s+3] - pose_origin[p_vis]) / s_safe[p_vis]
            )

    return out

def process_one(args):
    """Worker: reads one parquet, applies wrist-relative normalization, saves .npy."""
    root, rel_path, seq_id, cache_dir, seq_len, n_feat = args
    npy_path = f"{cache_dir}/{seq_id}.npy"
    if os.path.exists(npy_path):
        return True
    try:
        df_p = pd.read_parquet(
            f"{root}/{rel_path}",
            columns=["frame", "type", "landmark_index", "x", "y", "z"],
        )
        df_p = df_p[df_p["type"].isin(_USE_TYPES)].copy()
        if len(df_p) == 0:
            np.save(npy_path, np.zeros((seq_len, n_feat), dtype=np.float32))
            return True

        df_p[["x", "y", "z"]] = (
            df_p[["x", "y", "z"]].fillna(0.0).astype(np.float32)
        )

        frames    = sorted(df_p["frame"].unique())
        T         = len(frames)
        frame_map = {f: i for i, f in enumerate(frames)}
        df_p["fi"]   = df_p["frame"].map(frame_map).astype(np.int32)
        df_p["off"]  = df_p["type"].map(_TYPE_OFFSET)
        df_p["nlm"]  = df_p["type"].map(_TYPE_NLM)
        df_p         = df_p[df_p["landmark_index"] < df_p["nlm"]].copy()
        df_p["base"] = (df_p["off"] + df_p["landmark_index"] * 3).astype(np.int32)

        fi   = df_p["fi"].values
        base = df_p["base"].values

        # Direct assignment — avoids np.add.at duplicate accumulation bug
        raw = np.zeros((T, n_feat), dtype=np.float32)
        raw[fi, base]     = df_p["x"].values
        raw[fi, base + 1] = df_p["y"].values
        raw[fi, base + 2] = df_p["z"].values

        # Resample to seq_len
        if T == seq_len:
            out = raw
        elif T == 1:
            out = np.repeat(raw, seq_len, axis=0)
        else:
            idx  = np.linspace(0, T - 1, seq_len)
            lo   = np.floor(idx).astype(int)
            hi   = np.minimum(lo + 1, T - 1)
            frac = (idx - lo).reshape(-1, 1)
            out  = raw[lo] * (1 - frac) + raw[hi] * frac

        # ✅ Wrist-relative normalization (replaces per-sample z-score)
        out = normalize_landmarks(out)

        np.save(npy_path, out)
        return True
    except Exception:
        np.save(npy_path, np.zeros((seq_len, n_feat), dtype=np.float32))
        return False

def time_stretch(x: np.ndarray, rate: float) -> np.ndarray:
    """
    Stretch or compress the time axis then resample back to original length.
    rate > 1 → compress (fast signing)
    rate < 1 → stretch  (slow signing)
    """
    T   = x.shape[0]
    new_T = max(2, int(T / rate))
    idx   = np.linspace(0, T - 1, new_T)
    lo    = np.floor(idx).astype(int).clip(0, T - 1)
    hi    = np.minimum(lo + 1, T - 1)
    frac  = (idx - lo).reshape(-1, 1)
    stretched = x[lo] * (1 - frac) + x[hi] * frac
    # Resample back to T
    idx2  = np.linspace(0, new_T - 1, T)
    lo2   = np.floor(idx2).astype(int).clip(0, new_T - 1)
    hi2   = np.minimum(lo2 + 1, new_T - 1)
    frac2 = (idx2 - lo2).reshape(-1, 1)
    return (stretched[lo2] * (1 - frac2) + stretched[hi2] * frac2).astype(np.float32)


def augment(x: np.ndarray) -> np.ndarray:
    """
    Identity-preserving augmentations ONLY.

    Removed vs previous attempts:
      ✗ flip  — changes the ASL sign (different word or nonsense)
      ✗ scale — breaks wrist-relative normalization

    Applied here:
      ✓ time-stretch — simulates signing at different speeds
      ✓ Gaussian noise — simulates MediaPipe jitter / webcam noise
      ✓ frame masking — simulates partial occlusion / detection failures
    """
    x = x.copy()
    T = x.shape[0]

    # 1. Time stretch (most impactful for generalisation)
    if random.random() < 0.6:
        rate = random.uniform(0.75, 1.4)
        x = time_stretch(x, rate)

    # 2. Gaussian noise on coordinates
    if random.random() < 0.5:
        x += np.random.normal(0, 0.015, x.shape).astype(np.float32)

    # 3. Random frame masking (simulates dropped detection frames)
    if random.random() < 0.4:
        n_mask = random.randint(1, 5)
        mask_frames = np.random.choice(T, n_mask, replace=False)
        x[mask_frames] = 0.0

    return x


class ASLDataset(Dataset):
    def __init__(self, df_, augment_fn=None):
        self.rows = [
            (r.npy, int(r.label))
            for r in df_.itertuples()
            if os.path.exists(r.npy)
        ]
        self.aug = augment_fn

    def __len__(self):
        return len(self.rows)

    def __getitem__(self, i):
        path, lbl = self.rows[i]
        try:
            x = np.load(path)
            if self.aug is not None:
                x = self.aug(x)
            return torch.from_numpy(x), torch.tensor(lbl, dtype=torch.long)
        except Exception:
            return (
                torch.zeros(CFG.SEQ_LEN, CFG.N_FEAT, dtype=torch.float32),
                torch.tensor(lbl, dtype=torch.long),
            )

class SinPE(nn.Module):
    """Sinusoidal positional encoding."""
    def __init__(self, d, max_len, drop):
        super().__init__()
        self.drop = nn.Dropout(drop)
        pe  = torch.zeros(max_len, d)
        pos = torch.arange(max_len).float().unsqueeze(1)
        div = torch.exp(torch.arange(0, d, 2).float() * (-math.log(10000.0) / d))
        pe[:, 0::2] = torch.sin(pos * div)
        pe[:, 1::2] = torch.cos(pos * div)
        self.register_buffer("pe", pe.unsqueeze(0))   # (1, max_len, d)

    def forward(self, x):
        return self.drop(x + self.pe[:, :x.size(1)])


class ASLModel(nn.Module):
    def __init__(self, n_feat, n_cls, d, nh, nl, dff, drop, seq_len):
        super().__init__()

        # ── Conv1D stem ───────────────────────────────────────────────────────
        # Captures LOCAL finger-motion patterns before the Transformer sees
        # the global sequence.  Input: (B, n_feat, T), output: (B, d, T)
        self.stem = nn.Sequential(
            nn.Conv1d(n_feat, d, kernel_size=3, padding=1),
            nn.GELU(),
            nn.Conv1d(d, d, kernel_size=3, padding=1),
            nn.GELU(),
        )
        self.stem_norm = nn.LayerNorm(d)   # applied after permuting to (B,T,d)

        # ── CLS token + positional encoding ───────────────────────────────────
        self.cls = nn.Parameter(torch.randn(1, 1, d) * 0.02)
        self.pe  = SinPE(d, seq_len + 1, drop)   # +1 for CLS token

        # ── Transformer encoder ───────────────────────────────────────────────
        enc_layer = nn.TransformerEncoderLayer(
            d_model=d, nhead=nh, dim_feedforward=dff,
            dropout=drop, batch_first=True, norm_first=True,
        )
        self.encoder = nn.TransformerEncoder(enc_layer, num_layers=nl)

        # ── Classification head ───────────────────────────────────────────────
        self.head = nn.Sequential(
            nn.LayerNorm(d),
            nn.Dropout(drop),
            nn.Linear(d, n_cls),
        )

    def forward(self, x):                        # x : (B, T, n_feat)
        # Conv stem
        x = self.stem(x.permute(0, 2, 1))       # (B, d, T)
        x = self.stem_norm(x.permute(0, 2, 1))  # (B, T, d)

        # Prepend CLS token
        B   = x.size(0)
        cls = self.cls.expand(B, -1, -1)         # (B, 1, d)
        x   = torch.cat([cls, x], dim=1)         # (B, T+1, d)

        # Positional encoding + Transformer
        x = self.pe(x)
        x = self.encoder(x)

        # Classify using CLS token output (index 0)
        return self.head(x[:, 0])

# Linear warmup then cosine decay
def _lr_lambda(epoch):
    if epoch < CFG.WARMUP:
        return (epoch + 1) / CFG.WARMUP
    progress = (epoch - CFG.WARMUP) / max(CFG.EPOCHS - CFG.WARMUP, 1)
    return 0.5 * (1.0 + math.cos(math.pi * progress))

sched = optim.lr_scheduler.LambdaLR(opt, _lr_lambda)

def ckpt_dict(ep):

    return {

        "epoch":    ep,
        "best_acc": best_acc,
        "sign2idx": sign2idx,
        "idx2sign": {str(k): v for k, v in idx2sign.items()},
        "n_classes": N_CLS,
        "cfg": {
            "SEQ_LEN":  CFG.SEQ_LEN,
            "N_FEAT":   CFG.N_FEAT,
            "D_MODEL":  CFG.D_MODEL,
            "N_HEADS":  CFG.N_HEADS,
            "N_LAYERS": CFG.N_LAYERS,
            "D_FF":     CFG.D_FF,
        },
        "normalization": "wrist_relative_v5",
        "history": history,
    }

for ep in range(CFG.EPOCHS):
    t0 = time.time()
    model.train()
    tr_l = tr_c = tr_n = 0

    for bi, (feat, lbl) in enumerate(trn_dl):
        feat = feat.to(DEVICE, non_blocking=True)
        lbl  = lbl.to(DEVICE,  non_blocking=True)
        opt.zero_grad(set_to_none=True)

        if scaler:
            with torch.amp.autocast("cuda"):
                out  = model(feat)
                loss = crit(out, lbl)
            scaler.scale(loss).backward()
            scaler.unscale_(opt)
            nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            scaler.step(opt)
            scaler.update()
        else:
            out  = model(feat)
            loss = crit(out, lbl)
            loss.backward()
            nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            opt.step()

        tr_c += (out.argmax(1) == lbl).sum().item()
        tr_n += lbl.size(0)
        tr_l += loss.item() * lbl.size(0)

        if (bi + 1) % 100 == 0:
            log(f"  Ep{ep+1} [{bi+1}/{len(trn_dl)}] "
                f"loss={tr_l/tr_n:.4f} acc={tr_c/tr_n:.4f} "
                f"lr={opt.param_groups[0]['lr']:.2e}")

    sched.step()

    # ── Validation ────────────────────────────────────────────────────────────
    model.eval()
    vl_c = vl_n = t3c = 0
    with torch.no_grad():
        for feat, lbl in vln_dl:
            feat = feat.to(DEVICE)
            lbl  = lbl.to(DEVICE)
            if scaler:
                with torch.amp.autocast("cuda"):
                    out = model(feat)
            else:
                out = model(feat)
            vl_c += (out.argmax(1) == lbl).sum().item()
            vl_n += lbl.size(0)
            t3    = out.topk(3, 1).indices
            t3c  += sum(lbl[j].item() in t3[j].tolist() for j in range(len(lbl)))

    tr_acc = tr_c / max(tr_n, 1)
    vl_acc = vl_c / max(vl_n, 1)
    top3   = t3c  / max(vl_n, 1)
    gap    = tr_acc - vl_acc
    mins   = (time.time() - t0) / 60

    log(f"\nEP {ep+1}/{CFG.EPOCHS} ({mins:.1f} min) | "
        f"train={tr_acc:.4f} | val={vl_acc:.4f} | top3={top3:.4f} | "
        f"gap={gap:.4f} | lr={opt.param_groups[0]['lr']:.2e}")

    history.append({
        "ep": ep + 1,
        "train": round(tr_acc, 4),
        "val":   round(vl_acc, 4),
        "top3":  round(top3, 4),
        "gap":   round(gap, 4),
    })

    if vl_acc > best_acc:
        best_acc     = vl_acc
        patience_cnt = 0
        save_bundle()
        log(f"  ✅ BEST val_acc: {best_acc:.4f}")
    else:
        patience_cnt += 1
        log(f"  ⏳ {patience_cnt}/{CFG.PATIENCE}")
        if patience_cnt >= CFG.PATIENCE:
            log("  🛑 Early stopping.")
            break

    if (ep + 1) % 5 == 0 and vl_acc <= best_acc:
        save_bundle()   # periodic checkpoint even without improvement

    torch.cuda.empty_cache()
    gc.collect()


# ── Final save + push ──────────────────────────────────────────────────────────
